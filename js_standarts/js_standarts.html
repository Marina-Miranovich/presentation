JavaScript Standarts and best practices


// bad
var item = new Object();

// good
var item = {};
         

// bad
var items = new Array();

// good
var items = [];
         

// bad
var superman = {
  class: 'alien'
};

// bad
var superman = {
  klass: 'alien'
};

// good
var superman = {
  type: 'alien'
};
         

// bad
function nope(name, options, arguments) {
  // ...code...
}

// good
function yup(name, options, args) {
  // ...code...
}
         

// bad
var i, items = getItems(),
    dragonball,
    goSportsTeam = true,
    len;

// hmmmm...
var items = getItems();
var goSportsTeam = true;
var dragonball = 'z';

// mmmm :)
var items = getItems(),
    goSportsTeam = true,
    dragonball,
    length,
    i;
         

// bad
function() {
    if (typeof name === 'underfined') {
        var name = getName();
    }

  return name;
}

// good
function() {
    var name;

    if (typeof name === 'underfined') {
        name = getName();
    }

    return name;
}
         

// bad
if (test)
  return false;

// hmmmm....
if (test) return false;

// good
if (test) {
  return false;
}
         

// bad
function test()
{
  console.log('test');
}

// hmmmm...
function test(){
  console.log('test');
}

// good
function test() {
  console.log('test');
}
         

// bad
var once
  , upon
  , aTime;

// good
var once,
    upon,
    aTime;
         

  // bad
  var hero = {
    firstName: 'Kevin',
    lastName: 'Flynn',
  };

// good
  var heroes = [
    'Batman',
    'Superman'
  ];
         

var a = 2 + '5'; // '25';

// bad
var val = +inputValue;

// bad
var val = new Number(inputValue);

// bad
var val = parseInt(inputValue);
         

// good
var val = Number(inputValue);

// good
var val = parseInt(inputValue, 10);
         

var age = 0;

// bad
var hasAge = new Boolean(age);

// good
var hasAge = Boolean(age);

// good
var hasAge = !!age;
