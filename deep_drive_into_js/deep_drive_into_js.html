Functions

Function is an object that has a few hidden properties. it has a hidden link to the object prototype, function context (this) and and the callable code that realizes function's behavior.

Method call pattern


var myObj = {
	value: 0,
	getValue: function () {
		return this.value;
	}
};
myObj.getValue() //0

Function call pattern


var myObj = {
	value: 0,
	getValue: function () {
		var summ = function (num1, num2) {
			console.log('summ', this);  // window
			return (num1 + num2);
		}
		summ(2, 3);
		console.log('getValue', this); //myObj
		return this.value;
	}
};
myObj.getValue(); //0

Constructor call pattern


var Quo = function (status) {
	this.status = status;
};
var quo = new Quo();

Apply, Call and Bind call pattern


var context = { foo: "bar" };
var returnFoo = function () {
  return this.foo;
};
returnFoo(); //undefined

Apply, Call and Bind call pattern


returnFoo.call(context); // => bar
		

returnFoo.apply(context); // => bar
		

var bound = returnFoo.bind(context);
bound(); // => "bar"

Closure


var myObj = (function() {
	var value = 0;
	return {
		getValue: function () {
			return value;
		},
		incrementValue: function () {
			value ++;
		}
	};
	
})();

Closure


var show_numbers = function () {
	for (var i = 0; i < 10; i ++) {
		setTimeout(function(){
			console.log(i);
		}, 10);
	}
}
show_numbers(); // 10, 10, 10, ...

Closure


var helper = function (i) {
	return function () {
		console.log(i);
	}	
};
var show_numbers = function () {
	for (var i = 0; i < 10; i ++) {
		setTimeout(helper(i), 10);
	}
}
show_numbers(); // 0, 1, 2, ....
